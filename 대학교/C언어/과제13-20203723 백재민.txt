#include <stdio.h>
#include <stdlib.h>

void main()
{
	int n, i, sum = 0, * p;
	double avg;

	printf("학생 수를 입력하시오\n");
	scanf_s("%d", &n);

	p = (int*)malloc(n * sizeof(int));
	printf("학생수만큼의 점수를 입력하시오\n");
	for (i = 0; i < n; i++) {
		scanf_s("%d", p + i);
		sum += p[i];
	}
	avg = (double)sum / n;
	int max = p[0];
	for (i = 1; i < n; i++) {
		if (p[i] > max) max = p[i];
	}
	printf("평균: %.2lf 최고점: %d \n", avg, max);
}
